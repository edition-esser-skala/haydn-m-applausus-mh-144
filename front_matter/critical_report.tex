\begin{movement}{quidvideo}
  \voice[Amor vel Galathea]
  Quid video – Superi –
  quis splendor oculis sese meis objicit!
  Quis plaudens populus garrulum implet nemus!
  Quid! Charitum sacrata domus! Incensus focus!
  Quid! Damon ibi lunato poplite
  adorans virum Majestate praecelsum!

  \voice[Genius vel Damon]
  Fave excelse Praesul!
  Nutuque benigno excipe vota,
  quae subditus omnis pectore grato gerit.

  \voice[Galathea]
  Quid audio – Damon in vota profusus?
  Accedens junget socia vota Soror.
\end{movement}

\begin{movement}{vivedecus}
  \voice[Damon]
  Vive Decus Mæcenatum,
  vive Pater pauperum,
  vide Burae cor oblatum
  tibi loco munerum.
  Fave tuis quod precantur,
  fave Abbas clementissime,
  fave illos ne labantur,
  Pater Pastor optime.
\end{movement}

\begin{movement}{nunc}
  \voice[Damon]
  Nunc, Galathea, excelso Praesuli gratum prome melos,
  ut Charites suae edere vota queant.

  \voice[Galathea]
  Ast, Damon, vereor – ne posthaec carpas vile carmen:
  Esto tentare licet.
\end{movement}
